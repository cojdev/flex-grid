@mixin row($a:$gutter) {
    display: flex;
    @if $a != 0 {
        margin: 0 (-$a/2);
    }
}

@mixin column($a: 2, $b:$gutter) {
    flex-grow: $a;
    @if $b != 0 {
        margin: 0 $b/2;
    }
}

.row {
    @include row;
}

.col {
    @include column;
}
